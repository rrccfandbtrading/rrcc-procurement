📋 Google Sheets Setup Instructions

Follow these steps once to connect Google Sheets:

  1. Open Google Sheets and create a new spreadsheet.
  2. Name the spreadsheet: RRCC Procurement
  3. Go to Extensions → Apps Script
  4. Delete all existing code and paste the script below. Click the 💾 save icon.
  5. Click Deploy → New Deployment. For type, choose Web app.
  6. Set Execute as: Me and Who has access: Anyone ← this is the #1 reason sync fails!
  7. Click Deploy, then Authorize access (pick your Google account → Advanced → Go to project → Allow).
  8. Copy the Web app URL — it must end in /exec (NOT /dev).
  9. Paste it above, click 💾 Save URL, then 🔍 Test Connection.
⚠ If sync isn't working, check these in order:
FASTEST TEST: Copy your /exec URL, add ?action=ping to the end, and paste it into a NEW browser tab. You should see {"status":"ok","message":"pong"}. If you see a Google login/permission page instead → your deployment is NOT public (fix #1 below). If you see your own JSON, the script is fine.

1. Access = "Anyone" — re-open Deploy → Manage deployments → edit (pencil) → Who has access → Anyone. (Not "Anyone with Google account" — just "Anyone".)
2. Redeploy after ANY edit — Apps Script freezes a snapshot. After editing code: Deploy → Manage deployments → Edit → Version: New version → Deploy.
3. Use the /exec URL, not /dev (the /dev one only works while you're logged in as owner).
4. Authorize with the right account — when you deploy, authorize using the SAME Google account that owns the Sheet.
5. Run 🔍 Test Connection — it reads the response, and falls back to an iframe probe to tell you if the script is at least reachable.
Apps Script Code (copy all):
// RRCC Procurement — Google Sheets Sync v4 // GET (ping/pull) returns plain JSON; POST (push) reads the 'payload' form field. function doGet(e) { var p = (e && e.parameter) || {}; var action = p.action || 'ping'; var result; try { if (action === 'pull') result = pullAll(); else if (action === 'ping') result = {status:'ok', message:'pong', time:new Date().toISOString()}; else result = {status:'ok', message:'RRCC Sync Active'}; } catch(err) { result = {status:'error', message:String(err)}; } return reply(result, p.callback); } function doPost(e) { var p = (e && e.parameter) || {}; var result; try { // Client sends the JSON in a 'payload' form field (hidden-iframe POST) var raw = p.payload || (e.postData && e.postData.contents) || '{}'; var data = JSON.parse(raw); var action = data.action || p.action; if (action === 'sync') result = syncAll(data); else if (action === 'pull') result = pullAll(); else if (action === 'ping') result = {status:'ok', message:'pong'}; else result = {status:'error', message:'unknown action'}; } catch(err) { result = {status:'error', message:String(err)}; } return reply(result, p.callback); } function syncAll(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); writeSheet(ss, 'Suppliers', data.suppliers || [], ['id','name','contact','phone','email','address','notes','createdAt']); writeSheet(ss, 'Products', data.products || [], ['id','sku','name','category','supplierId','supplierName','cost','price','unit','status','createdAt']); writeSheet(ss, 'PurchaseOrders', data.pos || [], ['id','supplierId','supplierName','date','deliveryDate','status','vatMode','subtotal','vatAmount','totalCost','notes','items','createdAt']); writeSheet(ss, 'Settlements', data.settlements || [], ['id','supplierId','supplierName','poId','amount','method','reference','proofLink','proofNote','notes','date','createdAt']); writeSheet(ss, 'Categories', (data.categories||[]).map(function(c){return {name:c}}), ['name']); writeSheet(ss, 'CompanyInfo', [data.companyInfo || {}], ['name','tin','phone','email','address']); return {status:'ok', message:'synced', time:new Date().toISOString()}; } function pullAll() { var ss = SpreadsheetApp.getActiveSpreadsheet(); return { status:'ok', suppliers: readSheet(ss, 'Suppliers'), products: readSheet(ss, 'Products'), pos: readSheet(ss, 'PurchaseOrders'), settlements: readSheet(ss, 'Settlements'), categories: readSheet(ss, 'Categories').map(function(r){return r.name}).filter(Boolean), companyInfo: readSheet(ss, 'CompanyInfo')[0] || null }; } function writeSheet(ss, name, rows, headers) { var sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); sheet.clearContents(); var out = [headers]; rows.forEach(function(r){ out.push(headers.map(function(h){ var v = r[h]; return (v === null || v === undefined) ? '' : (typeof v === 'object' ? JSON.stringify(v) : v); })); }); sheet.getRange(1,1,out.length,headers.length).setValues(out); sheet.getRange(1,1,1,headers.length).setFontWeight('bold').setBackground('#6B21A8').setFontColor('white'); sheet.setFrozenRows(1); } function readSheet(ss, name) { var sheet = ss.getSheetByName(name); if (!sheet) return []; var data = sheet.getDataRange().getValues(); if (data.length < 2) return []; var headers = data[0]; return data.slice(1).filter(function(row){return row.some(function(c){return c!==''})}).map(function(row){ var obj = {}; headers.forEach(function(h, i){ var v = row[i]; if (h === 'items' && typeof v === 'string' && v.charAt(0) === '[') { try { v = JSON.parse(v); } catch(e) {} } obj[h] = v; }); return obj; }); } // Returns JSONP if a callback name is given, else plain JSON. function reply(obj, callback) { var json = JSON.stringify(obj); if (callback) { return ContentService.createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
